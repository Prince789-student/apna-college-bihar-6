rules_version = '2';
service cloud.firestore {
  match /databases/{database}/documents {
    // Users - Anyone can read profile, only owner or admin can write
    match /users/{uid} {
      allow read: if request.auth != null;
      allow write: if request.auth != null && (request.auth.uid == uid || get(/databases/$(database)/documents/users/$(request.auth.uid)).data.role == 'ADMIN');
    }

    // Announcements - Everyone can read, Only Admin can write/delete
    match /announcements/{annId} {
      allow read: if request.auth != null;
      allow write, delete: if request.auth != null 
        && get(/databases/$(database)/documents/users/$(request.auth.uid)).data.role == 'ADMIN';
    }

    // Default for other data (BEU Subjects, study tracker etc)
    match /{document=**} {
      allow read, write: if request.auth != null;
    }
  }
}
